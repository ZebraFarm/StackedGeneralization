# aX^2 + bX + c
true_model <- function(input, p = 1){
    scale = c(0,1,3) # (aX^2, bX)
    shift = 0      # c

    tmp = input %*% t(scale)
    ans = tmp[,1]^3 + tmp[,2]^2 + tmp[,3] + shift

    return(sum(ans))
}
